[Header(\"Configuraci\u00f3n de Movimiento\")]\n    [SerializeField] private float moveSpeed = 7f;\n    [SerializeField] private float jumpForce = 8f;\n\n    [Header(\"Detecci\u00f3n de Suelo\")]\n moveAction.Disable();\n        jumpAction.Disable();\n        pushAction.Disable();\n        jumpAction.performed -= context => TryJump();\n        pushAction.performed -= context => TryPush();\n    }\n\n    private void Update()\n    {\n        if (!IsOwner) return;\n\n        if (playerCam == null)\n        {\n            playerCam = Object.FindFirstObjectByType<Unity.Cinemachine.CinemachineCamera>();\n            if (playerCam != null)\n            {\n                playerCam.Follow = this.transform;\n            }\n        }\n\n        moveInput = moveAction.ReadValue<Vector2>();\n        isGrounded = Physics.CheckSphere(groundCheck.position, groundCheckRadius, groundLayer);\n    }\n\n    private void FixedUpdate()\n    {\n        if (!IsOwner) return;\n\n        Vector3 targetVelocity = new Vector3(moveInput.x * moveSpeed, rb.linearVelocity.y, moveInput.y * moveSpeed);\n        rb.linearVelocity = }\n    }\n\n    private void TryJump()\n    {\n        if (!IsOwner) return;\n        if (isGrounded)\n        {\n            rb.linearVelocity = new Vector3(rb.linearVelocity.x, 0f,
