"using System.Collections;\nusing Unity.Netcode;\nusing UnityEngine;\n\npublic class PlatformNode : NetworkBehaviour\n{\n    public enum PlatformState { Idle, Warning, Falling }\n    \n    [Header(\"Configuraci\u00c3\u00b3n Visual\")]\n    [SerializeField] private Renderer platformRenderer;\n    [SerializeField] private Color warningColor = Color.red;\n    [SerializeField] private Color[] platformColors;\n    private Color originalColor;\n\n    [Header(\"Configuraci\u00c3\u00b3n de F\u00c3\u00adsicas\")]\n    [SerializeField] private Rigidbody rb;\n    \n    // Sincroniza el estado en la red. Cuando cambie en el servidor, cambiar\u00c3\u00a1 en todos los clientes de forma autom\u00c3\u00a1tica.\n    private NetworkVariable<PlatformState> currentState = new NetworkVariable<PlatformState>(PlatformState.Idle);\n\n    private void Awake()\n    {\n        if (rb == null) rb = GetComponent<Rigidbody>();\n        if (platformRenderer == null) platformRenderer = GetComponent<Renderer>();\n\n        // Guardamos el color original (ej. el material de madera desgastada)\n        if (platformRenderer != null)\n        {\n            // Si pusimos colores en la lista, escoge uno al azar\n            if (platformColors.Length > 0)\n            {\n                originalColor = platformColors[Random.Range(0, platformColors.Length)];\n                platformRenderer.material.color = originalColor;\n            }\n            else\n            {\n                // Si la lista est\u00c3\u00a1 vac\u00c3\u00ada, guarda el color que ya ten\u00c3\u00ada\n                originalColor = platformRenderer.material.color;\n            }\n        }\n    }\n\n    public override void OnNetworkSpawn()\n    {\n        // Nos suscribimos al evento de cambio de valor para actualizar lo visual en los clientes\n        currentState.OnValueChanged += OnStateChanged;\n        ResetPlatform();\n    }\n\n    public override void OnNetworkDespawn()\n    {\n        // Limpieza de eventos al destruir el objeto\n        currentState.OnValueChanged -= OnStateChanged;\n    }\n\n    // Este m\u00c3\u00a9todo se ejecuta en todos los jugadores (Servidor y Clientes) cuando cambia currentState\n    private void OnStateChanged(PlatformState oldState, PlatformState newState)\n    {\n        switch (newState)\n        {\n            case PlatformState.Idle:\n                platformRenderer.material.color = originalColor;\n                break;\n            case PlatformState.Warning:\n                // Aqu\u00c3\u00ad cambia al color de alerta. Puedes expandir esto para activar un sistema de part\u00c3\u00adculas de chispas.\n                platformRenderer.material.color = warningColor;\n                break;\n            case PlatformState.Falling:\n                // Visualmente no requiere cambios dr\u00c3\u00a1sticos, las f\u00c3\u00adsicas del servidor se sincronizan solas.\n                break;\n        }\n    }\n\n    // Este m\u00c3\u00a9todo SOLO puede ser ejecutado por el Servidor/Host\n    public void TriggerCollapse(float warningDuration)\n    {\n        if (!IsServer) return;\n        StartCoroutine(CollapseSequence(warningDuration));\n    }\n\n    private IEnumerator CollapseSequence(float warningDuration)\n{\n    // 1. Cambiar a estado de advertencia\n    currentState.Value = PlatformState.Warning;\n    yield return new WaitForSeconds(warningDuration);\n\n    // 2. Cambiar a estado de ca\u00c3\u00adda\n    currentState.Value = PlatformState.Falling;\n    \n    // OJO: Agrega esta l\u00c3\u00adnea para soltar la plataforma en la ra\u00c3\u00adz de la escena\n    transform.SetParent(null); \n    \n    // Ahora las f\u00c3\u00adsicas funcionar\u00c3\u00a1n sin restricciones del padre\n    rb.isKinematic = false; \n}\n\n    private void ResetPlatform()\n    {\n        if (IsServer)\n        {\n            rb.isKinematic = true;\n            transform.rotation = Quaternion.identity;\n            currentState.Value = PlatformState.Idle;\n        }\n    }\n}\n"
